actor Patient
rectangle System {

    rectangle UI

  rectangle Internal {
    rectangle Controller
    rectangle Hardware
    rectangle Sensors
  }

UI -> Controller : Updates
Controller -> Hardware : Changes state
Sensors -> Controller : Updates
Patient <-left-> Hardware : Blood exchange
Patient -left-> Sensors : State change
